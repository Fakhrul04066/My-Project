#include <bits/stdc++.h>
using namespace std;

bool isConsonant(char c) {
    c = tolower(c);
    return (c >= 'a' && c <= 'z') && (c != 'a' && c != 'e' && c != 'i' && c != 'o' && c != 'u');
}

bool IsEasyPronunciation(string s) {
    int cnt = 0;
    for (int i = 0; i < (int)s.length(); i++) {
        if (isConsonant(s[i])) {
            cnt++;
            if (cnt >= 4) {
                return false;
            }
        } else {
            cnt = 0;
        }
    }
    return true; 
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        string s;
        cin >> s;
        if(s.length() < 4) {
            cout << "YES\n";
            continue;
        }   
        if (IsEasyPronunciation(s)) {
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }
    }
}
